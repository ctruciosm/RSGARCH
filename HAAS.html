Modelo de Haas (2004): R vs Julia

Dados Simulados

Simulamos dados utilizando R (funcao simulate() do pacote MSGARCH()).

library(MSGARCH)
spec <- CreateSpec()
omega_1 <- 0.18
alpha_1 <- 0.40
beta_1 <- 0.30
omega_2 <- 0.01
alpha_2 <- 0.20
beta_2 <- 0.70
p11 <- 0.70
p22 <- 0.90
# Sintaxe do pacote, os parametros devem entrar nessa ordem
par <- c(omega_1, alpha_1, beta_1, omega_2, alpha_2, beta_2, p11, 1 - p22)
set.seed(1234)
sim <- simulate(object = spec, nsim = 1L, nahead = 10000, nburn = 500L, par = par)
write.table(as.numeric(sim$draw[,1]), "msgarch_r_norm.csv", col.names = FALSE, row.names = FALSE)

Simulamos dados utilizando Julia (com a funcao simulate_haas() implementada) [utilizamos os mesmos parametros utilizados anteriormente]

using Distributions, Optim, Statistics, ForwardDiff, StatsFuns, DelimitedFiles
include("Julia/haas_dgp.jl")
simulate_haas (generic function with 1 method)

P = [0.7 0.3; 0.1  0.9];
n = 10000;
ω = [0.18, 0.01];
α  = [0.4, 0.2];
β = [0.3, 0.7];
burnin = 500;
distri = "norm";
Random.seed!(1234);
(r, h, s) = simulate_haas(n, distri, ω, α, β, P, burnin);
writedlm("msgarch_julia_norm.csv", r, ',');
